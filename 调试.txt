# ========================
# Git 分支开发流程
# ========================

# 1. 从 main 切新分支开发
cd "d:\files\demo\desk project-Paimon\paimon-desk-assistant"
git switch main
git pull
git switch -c feat/your-change-name

# 2. 开发完成后提交
git add .
git commit -m "your commit message"

# 3A. 如果满意，合并到 main
git switch main
git merge feat/your-change-name

# 3B. 如果不满意，回到 main，不合并
git switch main

# 4. 如果确定这个分支不要了，再删除
git branch -D feat/your-change-name
